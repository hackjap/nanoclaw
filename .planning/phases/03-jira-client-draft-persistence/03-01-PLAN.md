      title: string;
      description: string;
      issueType: 'Bug' | 'Task' | 'Story';
    }

    Internal `textToAdf(text: string): AdfDocument`:
    - Split on `\n\n`, filter empty strings
    - Map each paragraph to `{ type: 'paragraph', content: [{ type: 'text', text: para }] }`
    - Return `{ type: 'doc', version: 1, content: paragraphs }`

    Internal `getClient(): Version3Client`:
    - Lazy singleton. Read JIRA_HOST, JIRA_EMAIL, JIRA_API_TOKEN from readEnvFile.
    - Normalize host: strip trailing slashes.
    - Throw Error if any credential missing: `'Jira credentials not configured in .env (JIRA_HOST, JIRA_EMAIL, JIRA_API_TOKEN)'`
    - Create Version3Client with basic auth (email + apiToken).

    Exported `classifyError(err: unknown): JiraError`:
    - If `err instanceof HttpException`: check `err.cause?.response?.status` (or `(err as any).status`)
      - 401/403 -> `{ category: 'auth', userMessage: 'Jira 인증에 실패했습니다. 관리자에게 문의해주세요.', details: err.message }`
      - Other status -> `{ category: 'api', userMessage: 'Jira 이슈 생성에 실패했습니다: ' + err.message, details: 'HTTP ' + status + ': ' + err.message }`
    - Else (network/unknown): `{ category: 'network', userMessage: 'Jira 서버에 연결할 수 없습니다. 잠시 후 다시 시도해주세요.', details: err instanceof Error ? err.message : String(err) }`

    Exported `async createJiraIssue(draft: JiraDraftData): Promise<JiraCreateResult | JiraError>`:
    - Get client via getClient(). If throws, return classifyError of the error.
    - Build fields: `{ project: { key: JIRA_PROJECT_KEY }, summary: draft.title, issuetype: { name: draft.issueType }, description: textToAdf(draft.description) }`
    - Try `client.issues.createIssue({ fields })`.
    - On success: extract `result.key` and build URL as `${host}/browse/${result.key}`. Return `{ key: result.key, url }`.
    - On error: classify with classifyError.
      - If category === 'network': retry once. Log retry attempt. If retry also fails, return classifyError of retry error.
      - If category === 'auth' or 'api': return error immediately (no retry per D-14).

    Export `textToAdf` for testing (prefix with underscore convention or just export directly since tests need it).
