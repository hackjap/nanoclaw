    ```typescript
    // Types
    export interface ReactionTarget {
      channel: string;  // Slack channel ID (no 'slack:' prefix)
      timestamp: string; // Message ts to react on
    }

    export type ReactionStage = 'idle' | 'received' | 'processing' | 'completed' | 'failed';

    export const STAGE_EMOJI: Record<Exclude<ReactionStage, 'idle'>, string> = {
      received: 'eyes',
      processing: 'gear',
      completed: 'white_check_mark',
      failed: 'x',
    };

    export interface StatusReactionCallbacks {
      addReaction: (channel: string, ts: string, emoji: string) => Promise<void>;
      removeReaction: (channel: string, ts: string, emoji: string) => Promise<void>;
      sendStatusMessage?: (channel: string, threadTs: string, text: string) => Promise<void>;
    }

    // Stage ordering for forward-only transitions
    const STAGE_ORDER: Record<ReactionStage, number> = {
      idle: 0, received: 1, processing: 2, completed: 3, failed: 3,
    };
    ```

    StatusReaction class:
    - Constructor: takes `target: ReactionTarget`, `callbacks: StatusReactionCallbacks`, optional `statusMessageDelayMs: number` (default 10000 per research recommendation).
    - Private `currentStage: ReactionStage = 'idle'`
    - Private `statusMessageTimer: ReturnType<typeof setTimeout> | null = null`
    - Private `async transition(to: ReactionStage)`: checks `STAGE_ORDER[to] > STAGE_ORDER[currentStage]`, if not returns early (forward-only). Removes previous emoji via `callbacks.removeReaction` wrapped in try/catch + `logger.warn`. Adds new emoji via `callbacks.addReaction` wrapped in try/catch + `logger.warn`. Updates `currentStage`.
    - `async received()`: calls `transition('received')`
    - `async processing()`: calls `transition('processing')`. If `callbacks.sendStatusMessage` exists, starts a timer that calls `callbacks.sendStatusMessage(target.channel, target.timestamp, 'Working on it...')` after `statusMessageDelayMs`. Per D-05.
    - `async completed()`: cancels status message timer. calls `transition('completed')`
    - `async failed()`: cancels status message timer. calls `transition('failed')`
    - `cancelStatusMessage()`: clears the timer if set.

    Error handling: wrap each `addReaction`/`removeReaction` call individually in try/catch. Log with `logger.warn({ channel, ts, emoji, err }, 'Failed to add/remove reaction')`. Never re-throw. Per project convention (errors logged, not propagated).

    ```typescript
    /** Add an emoji reaction to a message. Idempotent: already_reacted is not an error. */
    async addReaction(channelId: string, ts: string, emoji: string): Promise<void> {
      try {
        await this.app.client.reactions.add({
          channel: channelId,
          timestamp: ts,
          name: emoji,
        });
      } catch (err: unknown) {
        if ((err as any)?.data?.error === 'already_reacted') return;
        logger.warn({ channelId, ts, emoji, err }, 'Failed to add reaction');
      }
    }

    /** Remove an emoji reaction from a message. Idempotent: no_reaction is not an error. */
    async removeReaction(channelId: string, ts: string, emoji: string): Promise<void> {
      try {
        await this.app.client.reactions.remove({
          channel: channelId,
          timestamp: ts,
          name: emoji,
        });
      } catch (err: unknown) {
        if ((err as any)?.data?.error === 'no_reaction') return;
        logger.warn({ channelId, ts, emoji, err }, 'Failed to remove reaction');
      }
    }
    ```
